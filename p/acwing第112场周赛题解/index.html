int a[N];
int ansl,ansr;
int main(){
	ios::sync_with_stdio(false),cin.tie(0);
	int n;
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>a[i];
	}
	for(int i=1;i<=n;i++){
		if(a[i]!=i){
			for(int j=1;j<=n;j++){
				if(a[j]==i){
					if(j>i){
						ansl=i,ansr=j;
					} 
					else ansl=j,ansr=i;
					break;
				}
			}
		}
	}
	int l=ansl,r=ansr;
	while(l<r){
		int t=a[l];
		a[l]=a[r];
		a[r]=t;
		l++;
		r--;
	}
	for(int i=1;i<=n;i++)if(a[i]!=a[i-1]+1){
		cout<<"0 0";
		return 0;
	}
	cout<<ansl<<" "<<ansr;
	return 0;
}
